<objective>
Wire the end-to-end ingestion pipeline: RSS connector (INGS-01) implementing the SourceConnector Protocol from Plan 02, the Redis Streams publisher with dedupe + XADD + Pub/Sub broadcast hint, the persister consumer group that writes to the signals hypertable, and the broadcaster consumer group that fans out to Pub/Sub for SSE subscribers (Plan 04). A worker entrypoint glues it together via APScheduler. This plan seeds one canonical RSS source (BBC World) via migration so `docker compose up` produces visible signal flow on first boot.

Purpose: This closes the loop "connector → Redis Stream → Postgres + Pub/Sub" that success criterion 6 in ROADMAP demands ("no connector writes directly to the DB"). The two-consumer-group split (persist + broadcast) is the load-bearing architectural choice from RESEARCH.md §Pattern 3 — collapsing them would couple DB latency to SSE latency and break the 5s SLA.

Output: Running worker container that polls the seeded RSS feed every 60s, normalizes items through the Protocol-bound connector, publishes to Redis Streams, persists to TimescaleDB, broadcasts to Pub/Sub. Verifiable by `redis-cli XLEN signals` > 0 and `SELECT count(*) FROM signals` > 0 after a polling cycle.
</objective>

    ```python
    from __future__ import annotations
    import hashlib
    import logging
    from datetime import datetime, timezone
    from typing import AsyncIterator
    import feedparser
    import httpx
    from sqlalchemy import select
    from sqlalchemy.dialects.postgresql import insert as pg_insert
    from src.connectors.base import SourceConnector, SourceHealth
    from src.db import async_session
    from src.models.feed_state import FeedState
    from src.schemas.signal import SignalEvent, Provenance

    log = logging.getLogger(__name__)


    def _parse_struct_time(st) -> datetime | None:
        if not st:
            return None
        try:
            return datetime(*st[:6], tzinfo=timezone.utc)
        except Exception:
            return None


    class RSSConnector:
        """Implements SourceConnector for RSS/Atom feeds. One instance per source_id."""

        def __init__(self, source_id: str, feed_url: str, poll_interval_s: int = 60) -> None:
            self.source_id = source_id
            self.feed_url = feed_url
            self.poll_interval_s = poll_interval_s

        async def _load_state(self) -> tuple[str | None, str | None]:
            async with async_session()() as s:
                fs = await s.scalar(select(FeedState).where(FeedState.source_id == self.source_id))
                if fs is None:
                    return None, None
                return fs.etag, fs.last_modified

        async def _save_state(self, etag: str | None, last_modified: str | None, success: bool, error: str | None = None) -> None:
            now = datetime.now(timezone.utc)
            async with async_session()() as s, s.begin():
                stmt = pg_insert(FeedState).values(
                    source_id=self.source_id, etag=etag, last_modified=last_modified,
                    last_fetch_at=now,
                    last_success_at=now if success else None,
                    last_error=None if success else error,
                ).on_conflict_do_update(
                    index_elements=[FeedState.source_id],
                    set_={
                        "etag": etag, "last_modified": last_modified,
                        "last_fetch_at": now,
                        **({"last_success_at": now, "last_error": None} if success else {"last_error": error}),
                    },
                )
                await s.execute(stmt)

        async def fetch(self) -> AsyncIterator[SignalEvent]:
            etag, last_modified = await self._load_state()
            headers: dict[str, str] = {}
            if etag: headers["If-None-Match"] = etag
            if last_modified: headers["If-Modified-Since"] = last_modified

            try:
                async with httpx.AsyncClient(timeout=30, follow_redirects=True) as client:
                    resp = await client.get(self.feed_url, headers=headers)
            except Exception as e:
                await self._save_state(etag, last_modified, success=False, error=str(e))
                raise

            if resp.status_code == 304:
                await self._save_state(etag, last_modified, success=True)
                return

            if resp.status_code >= 400:
                await self._save_state(etag, last_modified, success=False, error=f"HTTP {resp.status_code}")
                resp.raise_for_status()

            new_etag = resp.headers.get("etag") or etag
            new_mod = resp.headers.get("last-modified") or last_modified

            parsed = feedparser.parse(resp.content)
            if getattr(parsed, "bozo", 0):
                log.warning("feedparser bozo for %s: %s", self.source_id, getattr(parsed, "bozo_exception", ""))

            now = datetime.now(timezone.utc)
            for entry in parsed.entries:
                source_item_id = entry.get("id") or entry.get("link") or entry.get("title")
                if not source_item_id:
                    continue
                det_id = SignalEvent.make_id(self.source_id, source_item_id)
                ts = _parse_struct_time(entry.get("published_parsed")) or _parse_struct_time(entry.get("updated_parsed")) or now
                link = entry.get("link") or self.feed_url
                yield SignalEvent(
                    id=det_id,
                    source_item_id=str(source_item_id),
                    timestamp=ts,
                    author=entry.get("author"),
                    title=entry.get("title"),
                    content=entry.get("summary") or entry.get("description") or "",
                    tags=[t.get("term", "") for t in entry.get("tags", []) if isinstance(t, dict) and t.get("term")],
                    provenance=Provenance(
                        source_id=self.source_id,
                        source_type="rss",
                        source_url=link,
                        fetched_at=now,
                        ingested_at=now,
                        confidence=1.0,
                    ),
                    raw={k: v for k, v in entry.items() if isinstance(v, (str, int, float, bool, list, dict, type(None)))},
                )

            await self._save_state(new_etag, new_mod, success=True)

        async def health(self) -> SourceHealth:
            async with async_session()() as s:
                fs = await s.scalar(select(FeedState).where(FeedState.source_id == self.source_id))
            if fs is None:
                return SourceHealth(self.source_id, None, None, 0, 0.0, "degraded")
            status = "healthy" if (fs.last_error is None and fs.error_count_1h == 0) else ("degraded" if fs.error_count_1h < 5 else "failed")
            return SourceHealth(self.source_id, fs.last_success_at, fs.last_error, fs.error_count_1h, fs.throughput_per_min, status)
    ```

    ```python
    from __future__ import annotations
    import redis.asyncio as aioredis
    from src.schemas.signal import SignalEvent

    STREAM = "signals"
    BROADCAST_CHANNEL = "signals:new"
    DEDUPE_TTL_S = 86400
    STREAM_MAXLEN = 100_000


    async def publish(redis: aioredis.Redis, ev: SignalEvent) -> bool:
        """Publish a SignalEvent. Returns True if published, False if deduped."""
        # Pitfall 2: Redis dedupe key survives process restart.
        added = await redis.set(f"dedupe:{ev.id}", "1", nx=True, ex=DEDUPE_TTL_S)
        if not added:
            return False
        payload = ev.model_dump_json()
        await redis.xadd(STREAM, {"payload": payload}, maxlen=STREAM_MAXLEN, approximate=True)
        await redis.publish(BROADCAST_CHANNEL, payload)
        return True
    ```

    ```python
    from __future__ import annotations
    import time
    import redis.asyncio as aioredis


    def _throughput_key(source_id: str) -> str:
        return f"tput:{source_id}"


    async def record_fetch(redis: aioredis.Redis, source_id: str, ok: bool, items_emitted: int) -> None:
        now = int(time.time() * 1000)
        key = _throughput_key(source_id)
        # Each emitted item is a ZSET entry scored by ms; prune older than 60 s inside get_throughput.
        pipe = redis.pipeline()
        for i in range(max(items_emitted, 0)):
            pipe.zadd(key, {f"{now}-{i}": now})
        pipe.expire(key, 120)
        await pipe.execute()
        # Health: track a simple counter of errors in last hour.
        if not ok:
            err_key = f"errs:{source_id}"
            await redis.incr(err_key)
            await redis.expire(err_key, 3600)


    async def get_throughput(redis: aioredis.Redis, source_id: str, window_s: int = 60) -> float:
        now = int(time.time() * 1000)
        cutoff = now - window_s * 1000
        key = _throughput_key(source_id)
        # Prune old entries lazily
        await redis.zremrangebyscore(key, 0, cutoff)
        n = await redis.zcard(key)
        return float(n) / (window_s / 60.0)   # items per minute


    async def get_error_count_1h(redis: aioredis.Redis, source_id: str) -> int:
        v = await redis.get(f"errs:{source_id}")
        return int(v or 0)
    ```

    ```python
    from __future__ import annotations
    import asyncio
    import logging
    import redis.asyncio as aioredis
    from redis.exceptions import ResponseError
    from src.db import async_session
    from src.models.signal import Signal
    from src.pipeline.publisher import STREAM
    from src.schemas.signal import SignalEvent

    log = logging.getLogger(__name__)

    PERSIST_GROUP = "persist"
    PERSIST_CONSUMER = "persist-1"


    async def _ensure_group(redis: aioredis.Redis, group: str) -> None:
        try:
            await redis.xgroup_create(STREAM, group, id="0", mkstream=True)
        except ResponseError as e:
            if "BUSYGROUP" not in str(e):
                raise


    async def run_persister(redis: aioredis.Redis, stop: asyncio.Event | None = None) -> None:
        await _ensure_group(redis, PERSIST_GROUP)
        while stop is None or not stop.is_set():
            resp = await redis.xreadgroup(PERSIST_GROUP, PERSIST_CONSUMER, {STREAM: ">"}, count=100, block=5000)
            if not resp:
                continue
            for _stream, entries in resp:
                ids_to_ack: list[str] = []
                async with async_session()() as s, s.begin():
                    for entry_id, fields in entries:
                        try:
                            ev = SignalEvent.model_validate_json(fields["payload"])
                        except Exception as e:
                            log.error("persister: bad payload, dropping %s: %s", entry_id, e)
                            ids_to_ack.append(entry_id)
                            continue
                        row = Signal.from_event(ev)
                        await s.merge(row)   # idempotent upsert on composite PK
                        ids_to_ack.append(entry_id)
                # ACK only after commit
                if ids_to_ack:
                    await redis.xack(STREAM, PERSIST_GROUP, *ids_to_ack)
    ```

    ```python
    from __future__ import annotations
    import asyncio
    import logging
    import redis.asyncio as aioredis
    from redis.exceptions import ResponseError
    from src.pipeline.publisher import STREAM, BROADCAST_CHANNEL

    log = logging.getLogger(__name__)

    BROADCAST_GROUP = "broadcast"
    BROADCAST_CONSUMER = "broadcast-1"


    async def _ensure_group(redis: aioredis.Redis, group: str) -> None:
        try:
            await redis.xgroup_create(STREAM, group, id="0", mkstream=True)
        except ResponseError as e:
            if "BUSYGROUP" not in str(e):
                raise


    async def run_broadcaster(redis: aioredis.Redis, stop: asyncio.Event | None = None) -> None:
        """Consume signals stream and re-publish each payload to Pub/Sub for SSE subscribers.
        Ephemeral — if no subscribers, just ACK and move on (don't let the stream lag)."""
        await _ensure_group(redis, BROADCAST_GROUP)
        while stop is None or not stop.is_set():
            resp = await redis.xreadgroup(BROADCAST_GROUP, BROADCAST_CONSUMER, {STREAM: ">"}, count=100, block=5000)
            if not resp:
                continue
            for _stream, entries in resp:
                ids_to_ack: list[str] = []
                for entry_id, fields in entries:
                    payload = fields.get("payload")
                    if payload:
                        await redis.publish(BROADCAST_CHANNEL, payload)
                    ids_to_ack.append(entry_id)
                if ids_to_ack:
                    await redis.xack(STREAM, BROADCAST_GROUP, *ids_to_ack)
    ```

    ```python
    from src.redis_client import get_redis

    log = logging.getLogger(__name__)


    async def load_sources() -> None:
        clear()
        async with async_session()() as s:
            rows = (await s.scalars(select(Source).where(Source.enabled.is_(True)))).all()
        for row in rows:
            if row.source_type == "rss":
                feed_url = row.config.get("feed_url")
                if not feed_url:
                    log.warning("source %s has no feed_url; skipping", row.source_id)
                    continue
                register(RSSConnector(row.source_id, feed_url, poll_interval_s=row.config.get("poll_interval_s", 60)))
            else:
                log.warning("unknown source_type %s for %s (Phase 2+)", row.source_type, row.source_id)


    async def poll_once(connector) -> None:
        redis = get_redis()
        count = 0
        ok = True
        try:
            async for ev in connector.fetch():
                published = await publish(redis, ev)
                if published:
                    count += 1
        except Exception as e:
            ok = False
            log.exception("fetch failed for %s: %s", connector.source_id, e)
        await record_fetch(redis, connector.source_id, ok=ok, items_emitted=count)


    async def main() -> None:
        logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(name)s: %(message)s")
        get_settings()   # validate env early
        await load_sources()

        scheduler = AsyncIOScheduler(timezone="UTC")
        for c in all_connectors():
            scheduler.add_job(poll_once, "interval", seconds=c.poll_interval_s, args=[c], id=c.source_id, max_instances=1, coalesce=True)
        scheduler.start()
        log.info("scheduler started with %d connectors", len(all_connectors()))

        redis = get_redis()
        stop = asyncio.Event()
        # Run persister + broadcaster concurrently; keep process alive.
        await asyncio.gather(
            run_persister(redis, stop),
            run_broadcaster(redis, stop),
        )


    if __name__ == "__main__":
        asyncio.run(main())
    ```
